         * Per D-13: Store openid only (single-app mini program)
         */
        private String openid;

        private String nickname;

        @TableField("avatar_url")
        private String avatarUrl;

        private String phone;

        /**
         * Account balance in CNY
         */
        private BigDecimal balance;

        /**
         * Reward points for gift zone redemption
         */
        private Integer points;

        /**
         * Age verification status for alcohol sales compliance
         * Per D-05: Store age_verified boolean
         * Required by: 未成年人保护法, 酒类流通管理办法
         */
        @TableField("age_verified")
        private Boolean ageVerified;

        /**
         * Timestamp of age verification for audit trail
         * Per D-05: Store age_verified_at timestamp
         */
        @TableField("age_verified_at")
        private LocalDateTime ageVerifiedAt;

        /**
         * User status: 1=active, 0=disabled
         */
        private Integer status;

        @TableField(fill = FieldFill.INSERT)
        private LocalDateTime createdAt;

        @TableField(fill = FieldFill.INSERT_UPDATE)
        private LocalDateTime updatedAt;
    }
