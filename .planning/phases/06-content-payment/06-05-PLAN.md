                                            ContentAccessDecision access,
                                            List<PaymentMethodResponse> payMethods) {

      String preview = null;
      if (!access.isCanView() && Boolean.TRUE.equals(article.getPaid()) && article.getContent() != null) {

              ? article.getContent().substring(0, 150) + "..."
              : article.getContent();

              .id(article.getId())
              .title(article.getTitle())
              .type("article")
              .content(access.isCanView() ? article.getContent() : null)
              .preview(preview)  // NEW

              .price(price)

      if (userId == null) return false;
      Long count = contentLikeMapper.selectCount(
          new LambdaQueryWrapper<ContentLike>()
              .eq(ContentLike::getUserId, userId)
              .eq(ContentLike::getContentType, contentType)
              .eq(ContentLike::getContentId, contentId)
      );
      return count != null && count > 0;
  }
